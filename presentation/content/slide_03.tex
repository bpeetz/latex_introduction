	\begin{itemize}[<+-|alert@+>]
		\item Prozess:
		\begin{enumerate}
			\item Schreiben der .tex-Datei in Plaintext
			\\$\rightarrow$ dazu verschiedenste Programme möglich

			\item Kompilieren zum formatierten Dokument
		\end{enumerate}

		\item Verschiedenste Endprodukte möglich
		\item Nutzung v.a. bei Mischung von Formeln und Fließtext sinnvoll
	\end{itemize}

\end{frame}

\begin{frame}[fragile]{Syntax}
	\setbeamercolor{itemize/enumerate body}{fg=gray}

	\begin{itemize}[<+-|alert@+>]
		\item Befehle: \verb|\befehl[optionale Param.]{verpflichtende Param.}|, \\z.B. \verb|\textit{kursiver Text}|
		\item Umgebungen:\\ \verb|\begin{typ}| \dots \verb|\end{typ}|, \\z.B. \verb|\begin{enumerate}| \dots \verb|\end{enumerate}|
	\end{itemize}

\end{frame}

\begin{frame}[fragile]{Aufbau}
	\setbeamercolor{itemize/enumerate body}{fg=gray}
	
	Zwei Hauptteile:
	\begin{itemize}[<+-|alert@+>]
		\item \textit{preamble}
			\begin{itemize}
				\item \verb|\documentclass{}|
				\item Metadaten (z.B. \verb|\author|) $\rightarrow$ darstellen mit \verb|\maketitle|
				\item Teilweise: Globales Layout
				\item Definieren eigener Befehle mit \verb|\newcomand{}{}|
			\end{itemize}
		\item Inhaltsteil: \verb|\begin{document} … \end{document}|
			\begin{itemize}
				\item Strukturierung: z.B. \verb|\section{…}| $\rightarrow$ verschiedene Ebenen \\$\rightarrow$ Inhaltsverz. mit \verb|\tableofcontents|
				\item Quellen \& Bibliographie mit Packet \textit{bibtex}
				\item Eingliederung externer .tex-Dateien mit \verb|\input{}|
			\end{itemize}
	\end{itemize}
	
\end{frame}
